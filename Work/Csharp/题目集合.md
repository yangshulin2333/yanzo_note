#### 递归
```CSarp
#region 练习题1 
static void fun(int a)
{
    if (a == 11) { return; }
    Console.WriteLine(a);
    ++a;
    fun(a);
}
#endregion

#region 练习题2
static int fun2(int n) {

    if (n <= 1)
    {
        return 1;
    }
    else 
    {
        return n * fun2(n -1);
    }
}
#endregion


#region 练习题3
static int fun3(int n) {
    //结束条件
    if (n <= 1)
    {
        return 1;
    }
    else
    {
        //题目条件
       return  fun2(n) + fun3(n - 1);
    }

}
```
